\mhead{Research \newline Experience}%

\medskip
\textbf{Center for Global Change Science,} MIT, Cambridge, MA
\newline
\emph{Postdoctoral Associate} \rdate{2016-\textsc{present}}
\begin{itemize}[itemindent=-10pt]
\item Studying interactions between air quality, climate variability and climate change funded through an EPA Air Quality, Climate, and Energy Center grant, supervised by Noelle Selin and Susan Solomon
\item Analyzed large ensemble of IGSM/GEOS-Chem simulations to understand state-dependence of climate penalty calculations
\item Studied potential short-term policy impacts on air quality, health and the economy using a novel integrated assessment framework
\end{itemize}

\medskip
\textbf{Program in Atmospheres, Oceans, and Climate,} MIT, Cambridge, MA \newline
\emph{Research Assistant} \rdate{2011-2016}
\begin{itemize}[itemindent=-10pt]
\item Used novel uncertainty quantification techniques to develop emulator of droplet activation for parameterization in global models
\item Developed an open-source, modular parcel modeling framework for studying droplet activation from diverse aerosol populations and for evaluating activation schemes
\item Used global climate models (CESM, CMIP5 archive, AEROCOM Indirect Effects Experiment) to study aerosol indirect effects and aerosol-cloud interactions
\item Participated in Fifth Ice Nucleation Workshop Part 2 in Karlsruhe, Germany; assisted with the operation of the Spectrometer for Ice Nucleation (SPIN) and developed software for automating the instrument and designing experiment setups
\end{itemize}

\medskip
\textbf{Department of Earth and Atmospheric Sciences,} Cornell University, Ithaca, NY \newline
\emph{Undergraduate Research Assistant} \rdate{2008-2011}
\begin{itemize}[itemindent=-10pt]
	\item Studied biogeochemical/climate processes and interactions with a coupled carbon-climate model
	\item Performed and analyzed fully-coupled model simulations studying transient climate change in the 20th century
\end{itemize}

\medskip
\textbf{Center for Multiscale Modeling of Atmospheric Processes,} Colorado State University, \newline Fort Collins, CO \newline
\emph{Summer Intern} \rdate{\textsc{summer} 2010}
